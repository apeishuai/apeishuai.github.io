# 基本操作
sc query SNMP                  # 查询服务状态
sc start SNMP                   # 启动服务
sc stop SNMP                    # 停止服务
sc pause SNMP                   # 暂停服务
sc continue SNMP                # 恢复暂停的服务

# 配置操作
sc config SNMP start= auto       # 设置自动启动
sc config SNMP start= demand     # 设置手动启动
sc config SNMP start= disabled   # 设置禁用

# 查看服务配置
sc qc SNMP                       # 查询服务配置
sc enumdep SNMP                  # 查看服务依赖

# 远程操作
sc \\192.168.1.100 query SNMP    # 查询远程电脑的服务

# 基本操作
net start                        # 查看所有已启动的服务
net start "SNMP Service"         # 启动服务（注意引号）
net stop "SNMP Service"          # 停止服务
net pause "SNMP Service"         # 暂停服务
net continue "SNMP Service"      # 恢复服务

# 注意：net命令不支持配置修改

Get-WindowsCapability -Online -Name “SNMP*” //检查是否安装 Get-WindowsCapability -Online -Name “SNMP*” //安装snmp服务
